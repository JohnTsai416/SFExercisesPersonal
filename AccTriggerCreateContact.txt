trigger AccTrigger on Account (before insert, after insert) {
    
    if(Trigger.isBefore){
        for(Account a : Trigger.new){
            // Print ID before account is inserted 
            system.debug('Account name: ' + a.Id);
        }
    }
    // When an accout is inserted created a contact associated with that account 
    if(Trigger.isInsert){
        // List to store contacts to be created 
        List<Contact> newAccContact = new List<Contact>(); 
        // Iterate through accounts beign inserted       
        for(Account a : Trigger.new){
            // Create the contact match the Account ID with the field Account ID on Contact
            Contact newContact = new Contact(); 
            newContact.AccountId = a.ID;  
            // Add the new contact to the List 
            newAccContact.add(newContact); 
            
        }
        // Insert the List of Account after the loop has finished  
        insert newAccContact; 
    }
}
